Example query

SELECT o.office_id,
       COUNT(r.request_id) AS request_count
FROM offices o
JOIN requests r ON r.office_id = o.office_id
WHERE r.created_at >= DATE_SUB(CURDATE(), INTERVAL 30 DAY)
GROUP BY o.office_id
ORDER BY request_count DESC;
